```jsx
// @vitest-environment jsdom
import { describe, it, expect, beforeEach } from 'vitest';
import { render, screen, cleanup } from '@testing-library/react';
import userEvent from '@testing-library/user-event';
import * as engine from '../../../engine.js';

// Boot pattern: mirror studio-boot.test.jsx — set window.VcfEngine before
// the dynamic import so the .jsx file picks up the engine reference.
let VcfFleetSizer;

beforeEach(async () => {
  cleanup();
  window.VcfEngine = engine;
  // Dynamic import re-evaluates module top-level on each call when path is
  // identical; if your studio-boot.test.jsx caches it, mirror that exactly.
  ({ VcfFleetSizer } = await import('../../../vcf-design-studio-v9.jsx'));
});

// Build a fleet where the mgmt cluster has at least one T0 Gateway so the
// new T0 Uplinks sub-section's gate (cluster.t0Gateways.length > 0) is open.
function fleetWithT0() {
  const fleet = engine.createFleet();
  const cluster = fleet.instances[0].domains[0].clusters[0];
  // Use whichever helper exists; createT0Gateway is the public factory.
  cluster.t0Gateways = [engine.createT0Gateway()];
  return fleet;
}

function fleetWithoutT0() {
  const fleet = engine.createFleet();
  const cluster = fleet.instances[0].domains[0].clusters[0];
  cluster.t0Gateways = [];
  return fleet;
}

describe('M1.3 T0 Uplinks sub-section', () => {
  it('is hidden when cluster.t0Gateways is empty', () => {
    render(<VcfFleetSizer initialFleet={fleetWithoutT0()} />);
    expect(screen.queryByText(/T0 Uplink 1/i)).toBeNull();
    expect(screen.queryByText(/T0 Uplink 2/i)).toBeNull();
  });

  it('is visible when cluster.t0Gateways has at least one entry', () => {
    render(<VcfFleetSizer initialFleet={fleetWithT0()} />);
    expect(screen.getByText(/T0 Uplink 1/i)).toBeInTheDocument();
    expect(screen.getByText(/T0 Uplink 2/i)).toBeInTheDocument();
  });

  it('typing a VLAN into Uplink 1 updates the input value', async () => {
    const user = userEvent.setup();
    render(<VcfFleetSizer initialFleet={fleetWithT0()} />);
    // The input is labelled "VLAN" inside the "T0 Uplink 1" card. Use the
    // accessible name produced by the label-input wiring in the panel.
    const vlanInputs = screen.getAllByLabelText(/^VLAN$/i);
    // The first two are the T0 Uplink VLAN inputs (Uplink 1, then Uplink 2);
    // there may be other VLAN inputs further down the card.
    const uplink1Vlan = vlanInputs[0];
    await user.clear(uplink1Vlan);
    await user.type(uplink1Vlan, '1647');
    expect(uplink1Vlan.value).toBe('1647');
  });

  it('typing a Gateway into Uplink 1 updates the input value', async () => {
    const user = userEvent.setup();
    render(<VcfFleetSizer initialFleet={fleetWithT0()} />);
    const gatewayInputs = screen.getAllByLabelText(/^Gateway$/i);
    const uplink1Gateway = gatewayInputs[0];
    await user.clear(uplink1Gateway);
    await user.type(uplink1Gateway, '10.0.16.1');
    expect(uplink1Gateway.value).toBe('10.0.16.1');
  });

  it('non-numeric VLAN entry leaves the input empty (coerces to null in state)', async () => {
    const user = userEvent.setup();
    render(<VcfFleetSizer initialFleet={fleetWithT0()} />);
    const vlanInput = screen.getAllByLabelText(/^VLAN$/i)[0];
    await user.clear(vlanInput);
    // <input type="number"> ignores alphabetic input at the DOM level; the
    // onChange handler still fires with an empty string, which the panel's
    // coercion logic must map to vlan: null. Controlled input then re-renders
    // with value="" (since vlan ?? "" === "").
    await user.type(vlanInput, 'abc');
    expect(vlanInput.value).toBe('');
  });
});
```

Place this helper alongside the existing `update`, `updateT0Gateway`, etc. handlers in `ClusterCard`:

```jsx
const updateUplink = (i, patch) => {
  const uplinks = (cluster.networks?.uplinks ?? [
    { vlan: null, gateway: "" },
    { vlan: null, gateway: "" },
  ]).map((u, idx) => (idx === i ? { ...u, ...patch } : u));
  update({ networks: { ...cluster.networks, uplinks } });
};

const coerceVlan = (raw) => {
  if (raw === "" || raw == null) return null;
  const n = Number(raw);
  return Number.isFinite(n) ? n : null;
};
```

```jsx
// @vitest-environment jsdom
import { describe, it, expect, beforeEach } from 'vitest';
import { render, screen, cleanup } from '@testing-library/react';
import userEvent from '@testing-library/user-event';
import * as engine from '../../../engine.js';

let VcfFleetSizer;

beforeEach(async () => {

  const fleet = engine.createFleet();
  const cluster = fleet.instances[0].domains[0].clusters[0];

    cluster.edgeCluster = engine.createEdgeCluster();
  }
  return fleet;

    expect(ips[0].value).toBe('');
    expect(ips[1].value).toBe('');
    expect(ips[2].value).toBe('');
  });

  it('editing Node 1 does not mutate Node 2 inputs', async () => {
    const user = userEvent.setup();

    const ips = screen.getAllByLabelText(/Uplink [12] IP/i);

    await user.type(ips[0], '10.0.17.2/24');
    await user.clear(ips[1]);
    await user.type(ips[1], '10.0.17.3/24');
    expect(ips[2].value).toBe('');
    expect(ips[3].value).toBe('');
  });
```
