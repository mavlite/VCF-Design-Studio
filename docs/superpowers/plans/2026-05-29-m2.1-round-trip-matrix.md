# M2.1 Round-trip Coverage Matrix — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** A self-guarding round-trip matrix where every value-bearing model field is asserted to survive the JSON save/load path and (if workbook-mapped) the CSV cell-map path, with a meta-guard that fails CI when a new field escapes coverage.

**Architecture:** One kitchen-sink fleet (every factory invoked) is walked by a type-aware sentinel stamper that records `path → value`. The matrix test runs that fleet through JSON (`migrateFleet`) and CSV (`emit → parse → import`, 9.0 + 9.1) round-trips and asserts sentinels survive. A meta-guard asserts every leaf is accounted for (CSV-matrix ∪ non-workbook-allowlist ∪ structural-ignore).

**Tech Stack:** Node + Vitest (node environment). Engine under test: `engine.js` (CommonJS, `module.exports = VcfEngine`). No new dependencies.

**Spec:** `docs/superpowers/specs/2026-05-29-m2.1-round-trip-matrix-design.md`

**Conventions to honor:**
- Engine is imported `import VcfEngine from "../../engine.js"` then destructured (see `tests/unit/round-trip-depth.test.js`).
- Test files use `// @vitest-environment node` is NOT required (node is the default); component tests opt into jsdom, engine tests stay default.
- No attribution trailers in commits. Conventional commit messages.
- Helpers that are NOT tests must not match `**/*.test.js` (so they aren't collected as suites). Put them in `tests/helpers/`.

```js
// tests/helpers/sentinel-walk.js
// Type-aware sentinel stamping for round-trip coverage (M2.1).
//
// Walks a plain object/array tree and replaces each value-bearing leaf with
// a UNIQUE sentinel chosen by the leaf's kind, so a round-trip that drops or
// mis-routes the field shows up as a missing/wrong sentinel. Values for
// constrained fields (vlan / ip / cidr / number) are kept VALID because the
// workbook cell-map validates on apply — an out-of-range sentinel would be
// rejected and look like a (false) round-trip failure.

const IP_RE = /^\d{1,3}(\.\d{1,3}){3}$/;
const CIDR_RE = /^\d{1,3}(\.\d{1,3}){3}\/\d{1,2}$/;

// Deterministic per-path spread so sibling fields get distinct values.
function hashPath(path) {
  let h = 0;
  for (let i = 0; i < path.length; i++) h = (h * 31 + path.charCodeAt(i)) >>> 0;
  return h;
}

// A distinct-but-valid octet quartet derived from the path hash, avoiding
// 0/255 in the last octet and keeping the first octet in 10..172.
function ipFor(path) {
  const h = hashPath(path);
  const a = 10 + (h % 100);            // 10..109
  const b = (h >> 3) % 254;            // 0..253
  const c = (h >> 6) % 254;            // 0..253
  const d = 1 + ((h >> 9) % 250);      // 1..250
  return `${a}.${b}.${c}.${d}`;
}

export function sentinelFor(path, current) {
  if (typeof current === "boolean") return !current; // distinct from default
  if (typeof current === "number") {
    const h = hashPath(path);
    if (/vlan/i.test(path)) return 2 + (h % 4092);     // 2..4093, in VLAN range
    if (/mtu/i.test(path)) return 1500 + (h % 7500);   // 1500..8999
    return 1 + (h % 9999);                              // generic distinct int
  }
  if (typeof current === "string") {
    if (CIDR_RE.test(current)) {
      const h = hashPath(path);
      return `${ipFor(path).split(".").slice(0, 3).join(".")}.0/${16 + (h % 12)}`; // /16../27
    }
    if (IP_RE.test(current)) return ipFor(path);
    return `rt::${path}`;
  }
  return undefined; // null/undefined/object handled by walker
}

export function stampSentinels(root) {
  const sentinels = {};
  function walk(node, path) {
    if (Array.isArray(node)) {
      return node.map((v, i) => walk(v, path ? `${path}.${i}` : String(i)));
    }
    if (node && typeof node === "object") {
      const out = {};
      for (const k of Object.keys(node)) out[k] = walk(node[k], path ? `${path}.${k}` : k);
      return out;
    }
    if (node === null || node === undefined) return node;
    const s = sentinelFor(path, node);
    if (s === undefined) return node;
    sentinels[path] = s;
    return s;
  }
  const stamped = walk(root, "");
  return { stamped, sentinels };
}
```

```js
// tests/helpers/kitchen-sink-fleet.js
// Builds a maximally-populated fleet: every exported create*/new* factory is
// exercised in a reachable position, every empty array/toggle that factories
// leave blank is filled. This is the single source-of-truth "model surface"
// for the M2.1 round-trip matrix. Field VALUES are not set here — the
// sentinel walk (tests/helpers/sentinel-walk.js) stamps them.
import VcfEngine from "../../engine.js";

const {
  newFleet, newWorkloadDomain, newWorkloadCluster,
  newT0Gateway, createHostIpOverride, createEdgeNode,
  createPortgroupSlot, createVdsLag, createNetworkIpv6,
  createSupervisorDeployment, createFederationLocalManager, createFederationTier1,
} = VcfEngine;

// Fill the parts of a cluster that factories leave empty by default.
function enrichCluster(cluster) {
  // 2 per-host overrides (per-host IP editing surface, #96).
  cluster.hostOverrides = [createHostIpOverride(), createHostIpOverride()];
  // T0 gateway (Theme 4 / VCF-INV-060).
  cluster.t0Gateways = [newT0Gateway()];
  // Edge nodes inside the always-present edgeCluster.
  if (cluster.edgeCluster && Array.isArray(cluster.edgeCluster.nodes)) {
    cluster.edgeCluster.nodes = [createEdgeNode(), createEdgeNode()];
  }
  // VDS lags + ipv6 + portgroups on the networks block.
  if (cluster.networks) {
    cluster.networks.vds = cluster.networks.vds || [];
    cluster.networks.lags = [createVdsLag()];
    cluster.networks.ipv6 = createNetworkIpv6();
    cluster.networks.portgroups = [createPortgroupSlot()];
  }
  // Supervisor deployment extras (Theme 11 Deploy WLD).
  if (cluster.supervisorConfig) {
    cluster.supervisorConfig.enabled = true;
    cluster.supervisorConfig.deployment = createSupervisorDeployment();
  }
  return cluster;
}

export function buildKitchenSinkFleet({ vcfVersion = "9.1" } = {}) {
  const fleet = newFleet();
  fleet.vcfVersion = vcfVersion;
  fleet.version = "vcf-sizer-v9";

  // Federation: ensure GM extras + ≥1 local manager + ≥1 tier1.
  if (fleet.federationConfig) {
    fleet.federationConfig.enabled = true;
    fleet.federationConfig.localManagers = [createFederationLocalManager()];
    fleet.federationConfig.tier1s = [createFederationTier1()];
  }

  const inst = fleet.instances[0];
  // Add a workload domain alongside the default mgmt domain.
  inst.domains.push(newWorkloadDomain("WLD-01"));

  // Enrich every cluster in every domain.
  for (const dom of inst.domains) {
    if (dom.type === "workload" && dom.clusters.length === 0) {
      dom.clusters.push(newWorkloadCluster("wld-cluster-01"));
    }
    dom.clusters.forEach(enrichCluster);
  }

  return fleet;
}
```

```js
const {
  emitWorkbookCellMapCsv, parseWorkbookCellMap, importWorkbookCellMap,
} = VcfEngine;

function csvRoundTrip(stampedFleet, workbookVersion) {
  const csv = emitWorkbookCellMapCsv(stampedFleet, null, { workbookVersion });
  const { fleet } = importWorkbookCellMap(parseWorkbookCellMap(csv), { workbookVersion });
  return fleet;
}

// Returns the set of sentinel paths that survive the CSV round-trip.
function csvSurvivors(workbookVersion) {
  const base = buildKitchenSinkFleet({ vcfVersion: workbookVersion });
  const { stamped, sentinels } = stampSentinels(base);
  const rebuilt = csvRoundTrip(stamped, workbookVersion);
  const survived = [];
  for (const [path, expected] of Object.entries(sentinels)) {
    if (getPath(rebuilt, path) === expected) survived.push(path);
  }
  return { survived: survived.sort(), sentinels };
}
```
